3use crate::component::{Component, ComponentError, Context, Node};

5/// Input component
6#[derive(Debug)]
7pub struct Input {
8    /// Input type (text, password, email, etc.)
9    pub input_type: String,
10    /// Input value
11    pub value: String,
12    /// Input placeholder
13    pub placeholder: Option<String>,
14    /// Whether the input is disabled
15    pub disabled: bool,
16    /// Whether the input is required
17    pub required: bool,
18    /// Input label
19    pub label: Option<String>,
20    /// Input error message
21    pub error: Option<String>,
22    /// Input helper text
23    pub helper_text: Option<String>,
24    /// On change handler
25    pub on_change: Option<fn(String)>,
26}
27
28/// Input props
29#[derive(Debug, Clone)]
30pub struct InputProps {
31    /// Input type (text, password, email, etc.)
32    pub input_type: Option<String>,
33    /// Input value
34    pub value: String,
35    /// Input placeholder
36    pub placeholder: Option<String>,
37    /// Whether the input is disabled
38    pub disabled: Option<bool>,
39    /// Whether the input is required
40    pub required: Option<bool>,
41    /// Input label
42    pub label: Option<String>,
43    /// Input error message
44    pub error: Option<String>,
45    /// Input helper text
46    pub helper_text: Option<String>,
47    /// On change handler
48    pub on_change: Option<fn(String)>,
49}
50
51impl Default for Input {
52    fn default() -> Self {
53        Self {
54            input_type: "text".to_string(),
55            value: String::new(),
56            placeholder: None,
57            disabled: false,
58            required: false,
59            label: None,
60            error: None,
61            helper_text: None,
62            on_change: None,
63        }
64    }
65}
66
67impl Component for Input {
68    type Props = InputProps;
69
70    fn create(props: Self::Props, _context: Context) -> Self {
71        Self {
72            input_type: props.input_type.unwrap_or_else(|| "text".to_string()),
73            value: props.value,
74            placeholder: props.placeholder,
75            disabled: props.disabled.unwrap_or(false),
76            required: props.required.unwrap_or(false),
77            label: props.label,
78            error: props.error,
79            helper_text: props.helper_text,
80            on_change: props.on_change,
81        }
82    }
83
84    fn update(&mut self, props: Self::Props) -> Result<(), ComponentError> {
85        self.input_type = props.input_type.unwrap_or_else(|| self.input_type.clone());
86        self.value = props.value;
87        self.placeholder = props.placeholder;
88        self.disabled = props.disabled.unwrap_or(self.disabled);
89        self.required = props.required.unwrap_or(self.required);
90        self.label = props.label;
91        self.error = props.error;
92        self.helper_text = props.helper_text;
93        self.on_change = props.on_change;
94        Ok(())
95    }
96
97    fn as_any(&self) -> &dyn std::any::Any {
98        self
99    }
100
101    fn as_any_mut(&mut self) -> &mut dyn std::any::Any {
102        self
103    }
104
105    fn render(&self) -> Result<Vec<Node>, ComponentError> {
106        // For now, return an empty Vec since we're not yet using the Node system
107        // In a real implementation, this would return the actual DOM nodes
108        Ok(vec![])
109    }
110}
