1//! Event delegation system for Orbit UI framework
2//!
3//! This module provides an enhanced event handling system with:
4//! - Event bubbling and capturing
5//! - Event delegation up and down the component tree
6//! - Stop propagation and prevent default functionality
7
8use std::any::TypeId;
9use std::collections::HashMap;
10use std::fmt::Debug;
11use std::sync::{Arc, Mutex, RwLock};
12
13use crate::component_single::Node;
14use crate::events::Event;

16/// Type alias for event handler function
17type EventHandler = Box<dyn Fn(&mut dyn Event, &EventPropagation) + Send + Sync>;
18
19/// Type alias for handler storage map
20type HandlerMap = Arc<RwLock<HashMap<TypeId, Vec<EventHandler>>>>;
21
22/// Specifies the event propagation phase
23#[derive(Debug, Clone, Copy, PartialEq, Eq)]
24pub enum PropagationPhase {
25    /// Event is traveling down from parent to target (DOM capturing phase)
26    Capturing,
27
28    /// Event is at the target component
29    Target,
30
31    /// Event is traveling up from target to parent (DOM bubbling phase)
32    Bubbling,
33}
34
35/// Controls how an event propagates through the component tree
36#[derive(Debug, Clone)]
37pub struct EventPropagation {
38    /// Whether the event should continue propagating
39    pub stopped: bool,
40
41    /// Whether the default action should be prevented
42    pub default_prevented: bool,
43
44    /// The current propagation phase
45    pub phase: PropagationPhase,
46
47    /// The component that is the original target of the event
48    pub target_id: Option<usize>,
49
50    /// The component that is currently handling the event
51    pub current_target_id: Option<usize>,
52}
53
54impl EventPropagation {
55    /// Create a new event propagation
56    pub fn new(phase: PropagationPhase) -> Self {
57        Self {
58            stopped: false,
59            default_prevented: false,
60            phase,
61            target_id: None,
62            current_target_id: None,
63        }
64    }
65
66    /// Stop event propagation
67    pub fn stop_propagation(&mut self) {
68        self.stopped = true;
69    }
70
71    /// Prevent the default action
72    pub fn prevent_default(&mut self) {
73        self.default_prevented = true;
74    }
75
76    /// Check if propagation is stopped
77    pub fn is_propagation_stopped(&self) -> bool {
78        self.stopped
79    }
80
81    /// Check if default is prevented
82    pub fn is_default_prevented(&self) -> bool {
83        self.default_prevented
84    }
85}

87/// An enhanced event that includes propagation information
88pub struct DelegatedEvent<E: Event> {
89    /// The original event
90    pub event: E,
91
92    /// Propagation control
93    pub propagation: EventPropagation,
94}
95
96impl<E: Event> DelegatedEvent<E> {
97    /// Create a new delegated event
98    pub fn new(event: E, phase: PropagationPhase) -> Self {
99        Self {
100            event,
101            propagation: EventPropagation::new(phase),
102        }
103    }
104
105    /// Stop propagation of this event
106    pub fn stop_propagation(&mut self) {
107        self.propagation.stop_propagation();
108    }
109
110    /// Prevent the default action for this event
111    pub fn prevent_default(&mut self) {
112        self.propagation.prevent_default();
113    }
114}

116/// Type of callback for delegated events
117#[allow(dead_code)]
118type DelegatedEventCallback<E> = Box<dyn Fn(&mut DelegatedEvent<E>) + Send + Sync>;

120/// Event delegate manages capturing, targeting, and bubbling of events
121#[derive(Default)]
122pub struct EventDelegate {
123    /// Map of event type to callbacks registered for capturing phase
124    capturing_handlers: HandlerMap,
125
126    /// Map of event type to callbacks registered for bubbling phase
127    bubbling_handlers: HandlerMap,
128
129    /// Map of event type to callbacks registered for just that target
130    target_handlers: HandlerMap,
131
132    /// Component ID for identification during propagation
133    component_id: Option<usize>,
134
135    /// Parent delegate for bubbling events up
136    parent: Option<Arc<Mutex<EventDelegate>>>,
137
138    /// Child delegates for capturing events down
139    children: Vec<Arc<Mutex<EventDelegate>>>,
140}
141
142impl EventDelegate {
143    /// Create a new event delegate
144    pub fn new(component_id: Option<usize>) -> Self {
145        Self {
146            capturing_handlers: Arc::new(RwLock::new(HashMap::new())),
147            bubbling_handlers: Arc::new(RwLock::new(HashMap::new())),
148            target_handlers: Arc::new(RwLock::new(HashMap::new())),
149            component_id,
150            parent: None,
151            children: Vec::new(),
152        }
153    }
154
155    /// Set the parent delegate for bubbling events
156    pub fn set_parent(&mut self, parent: Arc<Mutex<EventDelegate>>) {
157        self.parent = Some(parent);
158    }
159
160    /// Add a child delegate for capturing events
161    pub fn add_child(&mut self, child: Arc<Mutex<EventDelegate>>) {
162        self.children.push(child);
163    }
164
165    /// Register a handler for an event type in the capturing phase
166    pub fn capture<E: Event + 'static>(
167        &self,
168        handler: impl Fn(&E, &EventPropagation) + Send + Sync + 'static,
169    ) {
170        let type_id = TypeId::of::<E>();
171        let mut handlers = self.capturing_handlers.write().unwrap();
172
173        let boxed_handler: EventHandler = Box::new(move |e, prop| {
174            if let Some(event) = e.as_any().downcast_ref::<E>() {
175                handler(event, prop);
176            }
177        });
178
179        handlers.entry(type_id).or_default().push(boxed_handler);
180    }
181
182    /// Register a handler for an event type in the bubbling phase
183    pub fn bubble<E: Event + 'static>(
184        &self,
185        handler: impl Fn(&E, &EventPropagation) + Send + Sync + 'static,
186    ) {
187        let type_id = TypeId::of::<E>();
188        let mut handlers = self.bubbling_handlers.write().unwrap();
189
190        let boxed_handler: EventHandler = Box::new(move |e, prop| {
191            if let Some(event) = e.as_any().downcast_ref::<E>() {
192                handler(event, prop);
193            }
194        });
195
196        handlers.entry(type_id).or_default().push(boxed_handler);
197    }
198
199    /// Register a handler for an event type for this target only
200    pub fn on<E: Event + 'static>(
201        &self,
202        handler: impl Fn(&E, &EventPropagation) + Send + Sync + 'static,
203    ) {
204        let type_id = TypeId::of::<E>();
205        let mut handlers = self.target_handlers.write().unwrap();
206
207        let boxed_handler: EventHandler = Box::new(move |e, prop| {
208            if let Some(event) = e.as_any().downcast_ref::<E>() {
209                handler(event, prop);
210            }
211        });
212
213        handlers.entry(type_id).or_default().push(boxed_handler);
214    }
215
216    /// Dispatch an event starting from this delegate
217    pub fn dispatch<E: Event + 'static>(&self, event: &E, target_id: Option<usize>) {
218        // First, we do the capturing phase from the root down to the target
219        let mut propagation = EventPropagation {
220            stopped: false,
221            default_prevented: false,
222            phase: PropagationPhase::Capturing,
223            target_id,
224            current_target_id: self.component_id,
225        };
226
227        self.dispatch_capturing(event, &mut propagation, target_id);
228
229        // Next, we handle the target phase
230        if !propagation.is_propagation_stopped() && self.component_id == target_id {
231            propagation.phase = PropagationPhase::Target;
232            propagation.current_target_id = self.component_id;
233            self.handle_event(event, &propagation);
234        }
235
236        // Finally, we do the bubbling phase from the target up to the root
237        if !propagation.is_propagation_stopped() {
238            propagation.phase = PropagationPhase::Bubbling;
239            self.dispatch_bubbling(event, &mut propagation, target_id);
240        }
241    }
242
243    // Internal method to handle capturing phase
244    fn dispatch_capturing<E: Event + 'static>(
245        &self,
246        event: &E,
247        propagation: &mut EventPropagation,
248        target_id: Option<usize>,
249    ) {
250        // Handle this component's capturing handlers
251        if !propagation.is_propagation_stopped() {
252            propagation.current_target_id = self.component_id;
253            self.handle_capturing_event(event, propagation);
254        }
255
256        // If this is the target, stop capturing phase
257        if self.component_id == target_id {
258            return;
259        }
260
261        // Otherwise, continue capturing phase down to children
262        for child in &self.children {
263            if let Ok(child) = child.lock() {
264                if !propagation.is_propagation_stopped() {
265                    child.dispatch_capturing(event, propagation, target_id);
266                }
267            }
268        }
269    }
270
271    // Internal method to handle bubbling phase
272    fn dispatch_bubbling<E: Event + 'static>(
273        &self,
274        event: &E,
275        propagation: &mut EventPropagation,
276        _target_id: Option<usize>,
277    ) {
278        // Handle this component's bubbling handlers
279        if !propagation.is_propagation_stopped() {
280            propagation.current_target_id = self.component_id;
281            self.handle_bubbling_event(event, propagation);
282        }
283
284        // Continue bubbling up to parent
285        if !propagation.is_propagation_stopped() {
286            if let Some(parent) = &self.parent {
287                if let Ok(parent) = parent.lock() {
288                    parent.dispatch_bubbling(event, propagation, _target_id);
289                }
290            }
291        }
292    }
293
294    // Internal method to handle events during capturing phase
295    fn handle_capturing_event<E: Event + 'static>(
296        &self,
297        event: &E,
298        propagation: &EventPropagation,
299    ) {
300        let type_id = TypeId::of::<E>();
301        if let Ok(handlers) = self.capturing_handlers.read() {
302            if let Some(handlers) = handlers.get(&type_id) {
303                let mut boxed_event = event.box_clone();
304                for handler in handlers {
305                    handler(boxed_event.as_mut(), propagation);
306                    if propagation.is_propagation_stopped() {
307                        break;
308                    }
309                }
310            }
311        }
312    }
313
314    // Internal method to handle events during bubbling phase
315    fn handle_bubbling_event<E: Event + 'static>(&self, event: &E, propagation: &EventPropagation) {
316        let type_id = TypeId::of::<E>();
317        if let Ok(handlers) = self.bubbling_handlers.read() {
318            if let Some(handlers) = handlers.get(&type_id) {
319                let mut boxed_event = event.box_clone();
320                for handler in handlers {
321                    handler(boxed_event.as_mut(), propagation);
322                    if propagation.is_propagation_stopped() {
323                        break;
324                    }
325                }
326            }
327        }
328    }
329
330    // Internal method to handle events at the target
331    fn handle_event<E: Event + 'static>(&self, event: &E, propagation: &EventPropagation) {
332        let type_id = TypeId::of::<E>();
333        if let Ok(handlers) = self.target_handlers.read() {
334            if let Some(handlers) = handlers.get(&type_id) {
335                let mut boxed_event = event.box_clone();
336                for handler in handlers {
337                    handler(boxed_event.as_mut(), propagation);
338                    if propagation.is_propagation_stopped() {
339                        break;
340                    }
341                }
342            }
343        }
344    }
345}

347/// Helper to build an event delegation tree from a component tree
348pub fn build_delegation_tree(
349    node: &Node,
350    parent_delegate: Option<Arc<Mutex<EventDelegate>>>,
351) -> Arc<Mutex<EventDelegate>> {
352    let delegate = Arc::new(Mutex::new(EventDelegate::new(Some(node.id_value()))));
353
354    // Set parent relationship
355    if let Some(parent) = parent_delegate {
356        if let Ok(mut delegate_mut) = delegate.lock() {
357            delegate_mut.set_parent(parent.clone());
358        }
359
360        if let Ok(mut parent_mut) = parent.lock() {
361            parent_mut.add_child(delegate.clone());
362        }
363    }
364
365    // Build delegation tree for children
366    for child in node.children() {
367        let child_delegate = build_delegation_tree(child, Some(delegate.clone()));
368
369        if let Ok(mut delegate_mut) = delegate.lock() {
370            delegate_mut.add_child(child_delegate);
371        }
372    }
373
374    delegate
